add_namespace = wsv_franzferdinand

### Franz Ferdinand killed
news_event = {
	id = wsv_franzferdinand.1
	title = wsv_franzferdinand.1.t
	desc = wsv_franzferdinand.1.d
	picture = news_event_archduke_dead
	
	major = yes
	
	trigger = {
		date > 1914.6.27
		date < 1914.9.1
		NOT = { has_global_flag = franzferdinand_murdered }
		SER = {
			exists = yes
        	is_puppet = no 
        	has_war = no
        }		
		AUH = {
			exists = yes
			is_puppet = no 
			has_war = no
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}

	immediate = {
		set_global_flag = franzferdinand_murdered
		hidden_effect = {
			AUH = {
				add_war_support = 0.3
				set_country_flag = murder_of_franz_ferdinand
				if = {
					limit = { is_in_faction_with = GER }
					GER = { country_event = { id = wsv_franzferdinand.2  days = 2 } }
				}
			}
			SER = {
				set_country_flag=murder_of_franz_ferdinand
				add_named_threat={
					threat=5
					name= franzferdiand_dead_tt
				}
			}
			# activate the notice for everyone
			every_country = {
				limit = {
					OR = {
						original_tag = AUH
						original_tag = SER
						original_tag = USA
						original_tag = CSA
						original_tag = CAN
						original_tag = MEX
						capital_scope = {
							is_on_continent = europe
						}
					}
				}
				set_country_flag = got_july_crisis_mission
				activate_mission = austrian_investigation_countdown_mission
			}
		}
	}
		
	option = {
		name = wsv_franzferdinand.1.a
		trigger = { tag = AUH }
	}
	option = {
		name = wsv_franzferdinand.1.b
		trigger = { NOT = { tag = AUH } }
	}
}
