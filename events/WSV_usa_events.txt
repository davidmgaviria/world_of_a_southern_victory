## Election of 1912 ##
country_event = {
	id = wsv_usa.1
	title = wsv_usa.1.t
	desc = wsv_usa.1.d
	#picture = report_event_deal_made

	fire_only_once = yes

	mean_time_to_happen = {	days = 4 }

    trigger = {
    	tag = USA
        date > 1912.11.1
        date < 1913.10.10
        USA = {
	        OR = {
	        	has_government = democratic
	        	has_government = communism
	        }
        }
    }

    #teddy roosevelt
	option = {
		name = wsv_usa.1.a
		custom_effect_tooltip = teddy_becomes_president_tt
    	hidden_effect = {
    		retire_country_leader = yes
    		set_politics = {
			    ruling_party = democratic
			    last_election = "1912.11.5"
			    election_frequency = 48
			    elections_allowed = yes
			}
	        create_country_leader = {
	            name = "Theodore Roosevelt"
	            desc = USA_theodore_roosevelt_DESC 
	            picture = "gfx/portraits/USA/country_leaders/USA_theodore_roosevelt.dds"
	            expire = "1935.1.1"
	            ideology = liberalism
	            traits = {}
	        }
	        if = {
	        	limit = { democratic < 0.51 }
	        	set_political_party = { ideology = democratic	popularity = 51	}
	        }
    	}
    	
    	ai_chance = {
			factor = 80

			modifier = {
				factor = 0.5
				communism > 0.5
			}
		}
	}

	#eugene debs
	option = {
		name = wsv_usa.1.b
		custom_effect_tooltip = eugene_becomes_president_tt
		hidden_effect = {
			retire_country_leader = yes
			set_politics = {
			    ruling_party = communism
			    last_election = "1912.11.5"
			    election_frequency = 48
			    elections_allowed = yes
			}
	        create_country_leader = { 
				name = "Eugene V. Debs" 
				desc = USA_eugene_v_debs_DESC 
				picture = "gfx/portraits/USA/country_leaders/USA_eugene_v_debs.dds" 
				expire = "1935.1.1" 
				ideology = lincolnism 
				traits = { } 
			}
			if = {
				limit = { communism < 0.51 }
				set_political_party = { ideology = communism	popularity = 51	}
			}
		}
			
    	ai_chance = {
			factor = 20

			modifier = {
				factor = 2
				communism > 0.3
			}
			modifier = {
				factor = 2
				communism > 0.5
			}
			modifier = {
				is_historical_focus_on = yes
				factor = 0
			}
		}
	}
}

		name = wsv_usa.6.a

		add_war_support = -0.1
		# haiti gets mad

		diplomatic_relation = { country = HAI   relation = guarantee  active = no }

		ai_chance = {
			factor = 40
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
			modifier = {
				factor = 1.5 
				USA = { has_army_manpower = { size > 1000000 } }
			}
			modifier = {
				factor = 2 
				USA = { has_army_manpower = { size > 1500000 } }
			}
		}
	}

	# ask germany for help
	option = {
		name = wsv_usa.6.c
		custom_effect_tooltip = haitian_crisis_german_mediation_tt
		GER = { country_event = { id = wsv_usa.7  days = 2 } }

		ai_chance = {
			factor = 100
		}
	}

	# ask britain for help
	option = {
		name = wsv_usa.6.e  #d is description
		custom_effect_tooltip = haitian_crisis_british_mediation_tt
		ENG = { country_event = { id = wsv_usa.8  days = 2 } }

		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
	}
}
