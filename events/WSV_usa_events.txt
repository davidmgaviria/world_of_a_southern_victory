news_event = {
	id = wsv_usa.1
	title = wsv_usa.1.t
	desc = wsv_usa.1.d
	picture = news_event_confederate_soldiers

    major = yes

	mean_time_to_happen = {
		days = 0
	}

	trigger = {
		tag = USA
    	CSA = { has_war_with = USA }
        361 = { is_controlled_by = CSA } 
        361 = { is_owned_by = USA }
        NOT = { CSA = { has_country_flag = dc_taken } }
    }

    immediate = { 
    	hidden_effect = { CSA = { set_country_flag = dc_taken } } 
    }
	
	#country that took city and allies
	option = {
		name = wsv_usa.1.a
		trigger = { 
			has_war_with = USA
		}
	}

	#country that lost city and allies
	option = {
		name = wsv_usa.1.b
		trigger = {
			OR = {
				tag = USA
				is_in_faction_with = USA
			}
		}
	}

	#neutral countries
	option = {
		name = wsv_usa.1.c
		trigger = { 
			NOT = { is_in_faction_with = USA }
			NOT = { has_war_with = USA }
			NOT = { tag = USA }
		}
	} 
}
