SER_increase_army = {		# massively increase size
	allowed = {
		original_tag = SER
		has_start_date < 1937.1.1
	}
	enable = {
		has_army_manpower = { size < 200000 }
		has_war = no 
	}
	abort = {
		# pasue when about to start balkan war
		has_country_flag = balkan_league_member
		has_army_manpower = { size > 140000 }
	}

	ai_strategy = {
    	type = force_build_armies
    	value = 100
	}
	ai_strategy = {
	    type = build_army
	    id = infantry
	    value = 25
	}
	ai_strategy = {
		type = equipment_stockpile_surplus_ratio
		value = -10 	# reduce while we build up in case of issues
	}
}

SER_slow_against_AUH = {
	allowed = {
		original_tag = SER		
		has_start_date < 1937.1.1
	}
	enable = {
		has_war_with = AUH
		stockpile_ratio = {	 ## 3% of the armies equipment must be in reserves
			archetype = infantry_equipment
			ratio < 0.03
		}
	}
	abort = {
		OR = {
			stockpile_ratio = {	## 7% of the armies equipment must be in reserves
				archetype = infantry_equipment
				ratio > 0.07
			}
			surrender_progress > 0.15
		}		
	}

	ai_strategy = {
		type = front_control
		tag = AUH
		ratio = 0.6
		priority = 1
		ordertype = front
		execution_type = balanced
		execute_order = no
		manual_attack = yes
	}
}
