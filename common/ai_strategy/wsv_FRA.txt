FRA_launch_offensives_against_GER = {
	allowed = {
		original_tag = FRA		
		has_start_date < 1937.1.1
	}
	enable = {
		has_war_with = GER
		has_army_size = { size > 1600000 }   # wait to build up army	
		# launch offensives every 6 months (alongisde France)
		set_temp_variable = { days = global.num_days }
		modulo_temp_variable = { days = 180 }		# cycles of 6 months means 2 months combat plus 4 months pause
		check_variable = { days < 60 }
	}
	abort = {
		OR = {
			surrender_progress > 0.15	
			has_idea = FRA_halt_army_offensives
		}
	}

	ai_strategy = {
		type = front_control
		tag = GER
		ratio = 0.7
		priority = 25
		ordertype = front
		execution_type = careful
		execute_order = yes
		manual_attack = yes
	}

	ai_strategy = {
		type = consider_weak
		id = GER
		value = -15
	}
}
