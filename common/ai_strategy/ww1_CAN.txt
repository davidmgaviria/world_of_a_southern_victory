	}

	    type = build_army

	    value = 40
	}
}


CAN_defend_against_usa = {
	enable = {
		tag = CAN
		USA = { exists = yes }
		NOT = {
			is_in_faction_with = USA
			has_non_aggression_pact_with = USA
		}
		has_start_date < 1937.1.1
		threat > 0.15
	}
	abort_when_not_enabled = yes

	ai_strategy = {
		type = prepare_for_war
		id = "USA"
		value = 100
	}

	ai_strategy = {
		type = invade
		id = "USA"
		value = -1000
	}
	
	ai_strategy = {
		type = 	spare_unit_factor		
		value = -90
	}
}	


CAN_defensive_if_lack_of_guns_against_USA = {
	allowed = {
		original_tag = CAN		
		has_start_date < 1937.1.1
	}
	enable = {
		has_war_with = USA
		stockpile_ratio = {	## 5% of the armies equipment must be in reserves
			archetype = infantry_equipment
			ratio < 0.05
		}
		# NOT = {
		# 	has_global_flag = WW1_is_stuck
		# }
	}
	abort = {
		OR = {
			stockpile_ratio = {	## 7% of the armies equipment must be in reserves
				archetype = infantry_equipment
				ratio > 0.07
			}
			surrender_progress > 0.15
		}		
	}

	ai_strategy = {
		type = front_control
		tag = USA
		ratio = 0.5
		priority = 1
		ordertype = front
		execution_type = careful
		execute_order = no
		manual_attack = yes
	}
}
