# Increases level of conscription (written backwards to prevent chain effect)
increase_conscription_tier_effect = {
	if = {
		limit = { has_idea = all_adults_serve }
		add_ideas = scraping_the_barrel 
	}
	if = {
		limit = { has_idea = service_by_requirement }
		add_ideas = all_adults_serve 
	}
	if = {
		limit = { has_idea = extensive_conscription }
		add_ideas = service_by_requirement
	}
	if = {
		limit = { has_idea = limited_conscription }
		add_ideas = extensive_conscription
	}
	if = {
		limit = { OR = { has_idea = volunteer_only  has_idea = disarmed_nation } }
		add_ideas = limited_conscription
	}
}

# scope is state
CSA_black_rebels_escalate_state_effect = {
	if = {
		limit = { has_dynamic_modifier = { modifier = CSA_red_stronghold } }
		damage_building = {
		    type = infrastructure
		    damage = 0.5
		}
		damage_building = {
		    type = industrial_complex
		    damage = 0.5
		}
		add_manpower = -1000
	}
	if = {
		limit = { has_dynamic_modifier = { modifier = CSA_red_enclave } }
		hidden_effect = { 
			remove_dynamic_modifier = { modifier = CSA_red_enclave } 
		}
		add_dynamic_modifier = { modifier = CSA_red_stronghold }
	}
	if = {
		limit = { has_dynamic_modifier = { modifier = CSA_red_insurgents } }
		hidden_effect = { 
			remove_dynamic_modifier = { modifier = CSA_red_insurgents } 
		}
		add_dynamic_modifier = { modifier = CSA_red_enclave }
	}
}
# scope is state
CSA_black_rebels_pacify_state_effect = {
	if = {
		limit = { has_dynamic_modifier = { modifier = CSA_red_insurgents } }
		remove_dynamic_modifier = { modifier = CSA_red_insurgents } 
		clr_state_flag = contains_red_rebels
	}
	if = {
		limit = { has_dynamic_modifier = { modifier = CSA_red_enclave } }
		remove_dynamic_modifier = { modifier = CSA_red_enclave } 
		add_dynamic_modifier = { modifier = CSA_red_insurgents }
	}
	if = {
		limit = { has_dynamic_modifier = { modifier = CSA_red_stronghold } }
		remove_dynamic_modifier = { modifier = CSA_red_stronghold } 
		add_dynamic_modifier = { modifier = CSA_red_enclave }
	}
}

update_latin_american_exploitation_effect = {

	# first check csa's edge on influence
	set_temp_variable = { temp_influence_edge = CSA.csa_influence_in_caribbean }
	subtract_from_temp_variable = { temp_influence_edge = USA.usa_influence_in_caribbean }
	# update csa influence edge
	set_variable = { CSA.csa_influence_edge = temp_influence_edge }

	# clear first
	CSA = { remove_ideas = CSA_caribbean_empire_1 }
	CSA = { remove_ideas = CSA_caribbean_empire_2 }
	CSA = { remove_ideas = CSA_caribbean_empire_3 }
	# update accordingly 
	if = {
		limit = { check_variable = { var = CSA.csa_influence_edge   value = 30  compare = greater_than_or_equals } }
		CSA = { add_ideas = CSA_caribbean_empire_3 }
	}
	else_if = {
		limit = { check_variable = { var = CSA.csa_influence_edge   value = 20  compare = greater_than_or_equals } }
		CSA = { add_ideas = CSA_caribbean_empire_2 }
	}
	else_if = {
		limit = { check_variable = { var = CSA.csa_influence_edge   value = 10  compare = greater_than_or_equals } }
		CSA = { add_ideas = CSA_caribbean_empire_1 }
	}
}
