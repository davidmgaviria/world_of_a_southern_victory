					FRA = { add_to_faction = BEL }
				}
				if = {
					limit = {
						FRA = { is_faction_leader = yes }
						LUX = { is_in_faction = no   is_guaranteed_by = FRA }
					}
					FRA = { add_to_faction = LUX }
				}

		icon = generic_construction

		available = {
			controls_province = 3547

			factor = 100

			tag = GER

	# Germany throws in the towel
	greatwar_germany_offers_peace = {
		icon = generic_form_nation

		available = {
			always = yes
			OR = {
				casualties > 500000
				surrender_progress > 0.5
			}
		}

		visible = {
			tag = GER
			OR = {
				has_war_with = FRA
				has_war_with = RUS
			}
			has_start_date < 1937.1.1
		}

		days_re_enable = 90

		complete_effect = {
			if = {
				limit = {
					has_war_with = FRA
					FRA = { is_faction_leader = yes }

				FRA = { country_event = { id = wsv_peace_treaty.20  days = 1 } }
			}
			else_if = {
				limit = {
					has_war_with = RUS
					RUS = { is_faction_leader = yes }

			}
			custom_effect_tooltip = versailles_treaty_tt
		}

			## Getting pushed back is bad
			modifier = {

			}
			# Extreme casualties is also bad
			modifier = {
				add = 1
				casualties > 2000000
			}

			# Litte progress against France while being starved is bad

				add = 1
				has_idea = GER_blockade_effects

					AND = {

					}
				}
			}	
			# Litte progress against Russia while being starved is bad
			modifier = {
				add = 1
				has_idea = GER_blockade_effects  
				OR = {
					AND = {
						has_war_with = RUS
						RUS = { surrender_progress < 0.1 }
					}
				}
			}

			# Austria-Hungary surrendering is really bad
			modifier = {
				add = 2
				OR = {
					has_global_flag = auh_lost_greatwar
					AND = {
						is_in_faction_with = AUH

					}
				}
			}	

			# If we're completey losing, give up
			modifier = {
				add = 10
				surrender_progress > 0.55
			}	
		}
