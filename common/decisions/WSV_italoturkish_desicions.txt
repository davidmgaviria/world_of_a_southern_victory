	italoturkish_expedition_deadline = {

		icon = generic_operation

		allowed = {
			tag = ITA
		}

		visible = {
			has_country_flag = italian_libya_war

				add = 1

				    opponent = TUR
				    thousands > 5
				}
			}

				factor = 2

				    opponent = TUR

				}

				factor = 5
