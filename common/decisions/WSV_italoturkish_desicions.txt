italoturkish_war_decisions = {
	
	# Italian expedition timer
	italoturkish_expedition_deadline = {

		icon = generic_operation

		allowed = {
			tag = ITA
		}

		visible = {
			has_war_with = TUR
			NOT = {
				has_country_flag = ITA_Lib_War_End_Lost_States	
				has_country_flag = ITA_Lib_War_End_Victory		
			}
		}

		days_remove = 60

		fire_only_once = yes

		remove_effect = {
			set_country_flag = italoturkish_invasion_failed
			add_war_support = -0.1
			add_stability = -0.05
			add_political_power = -50
			#event telling us we suck
		}

		ai_will_do = {
			factor = 0
		}
	}


	# Italy throws in the towel
	italoturkish_italy_offers_peace = {

		icon = generic_form_nation

		allowed = {
			tag = ITA
		}

		available = {
			NOT = { has_decision = italoturkish_expedition_deadline }
			casualties > 10000
		}

		visible = {
			has_war_with = TUR
			NOT = {
				has_country_flag = ITA_Lib_War_End_Lost_States	
				has_country_flag = ITA_Lib_War_End_Victory		
			}
		}

		days_remove = 4

		days_re_enable = 60

		remove_effect = {
			TUR = { country_event = ww1_italianturkish.12 }
		}

		ai_will_do = {
			factor = 0

			modifier = {
				add = 1
				NOT = {
					controls_state = 448
					controls_state = 450
					controls_state = 451
					controls_state = 661
					controls_state = 662
					controls_state = 663
				}
			}

			modifier = {
				add = 1
				casualties > 50000
			}
		}
	}


	# Turkey throws in the towel
	italoturkish_turkey_offers_peace = {

		icon = generic_form_nation

		allowed = {
			tag = TUR
		}

		available = {
			OR = {
				casualties_inflicted_by = {
				    opponent = ITA
				    thousands > 50000
				}
				AND = {
					ITA = {
						controls_state = 448
						controls_state = 450
						controls_state = 451
					}
				}
			}
		}

		visible = {
			has_war_with = ITA
			ITA = {
				NOT = {
					has_country_flag = ITA_Lib_War_End_Lost_States	
					has_country_flag = ITA_Lib_War_End_Victory		
				}
			}
		}

		days_remove = 4

		days_re_enable = 60

		remove_effect = {
			ITA = { country_event =ww1_italianturkish.4 }
		}

		
		ai_will_do = {
			factor = 0

			#if we've lost everything give up
			modifier = {
				add = 1
				NOT = {
					controls_state = 448
					controls_state = 450
					controls_state = 451
					controls_state = 661
					controls_state = 662
					controls_state = 663
				}
			}

			#libya isnt worth the blood
			modifier = {
				add = 1
				casualties_inflicted_by = {
				    opponent = ITA
				    thousands > 80000
				}
			}
		}
	}
}
