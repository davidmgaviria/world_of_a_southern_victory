secondbalkan_war_decisions = {

	## Bulgaria war countdown
	secondbalkan_countdown_to_war = {
		icon = generic_ignite_civil_war

		allowed = {
			original_tag = BUL
		}

		activation = {
			has_global_flag = criticized_london_peace_conference
		}
		available = {
			has_country_flag = secondbalkan_war_prep_over
		}

		days_mission_timeout = 24
		is_good = no
		
		timeout_effect = {
			set_country_flag = secondbalkan_war_prep_over
			create_faction = "Greater Bulgaria"
			declare_war_on = {
				target = SER
				type = take_claimed_state
			}
			GRE = {
				if = {
					limit = { is_in_faction_with = SER }
					add_to_war = {
						targeted_alliance = SER
						enemy = BUL
					}
				}
			}
			hidden_effect = {
				news_event = wsv_secondbalkan.6
			}
		}
	}

	
	# Bulgaria surrenders
	secondbalkan_bulgaria_offers_peace = {
		icon = generic_form_nation

		allowed = {
			original_tag = BUL
		}
		visible = {
			has_global_flag = secondbalkan_war
			NOT = { has_global_flag = secondbalkanwar_over }
			has_war_with = SER
		}
		available = {
			OR = {
				casualties > 50000
				surrender_progress > 0
			}
		}

		days_remove = 3

		remove_effect = {
			country_event = wsv_secondbalkan.11
		}

		ai_will_do = {
			factor = 0
			# macedonia isnt worth this much if we've already lost it
			modifier = {
				add = 1
				casualties > 15000
			}
			# if we're being beaten surrender
			modifier = {
				add = 10
				surrender_progress > 0.3
			}
		}
	}


	# Anti-Bulgarian coalition surrenders
	secondbalkan_serbia_offers_peace = {
		icon = generic_form_nation

		allowed = {
			OR = {
				original_tag = SER
				original_tag = GRE
				original_tag = ROM
			}
		}
		visible = {
			has_global_flag = secondbalkan_war
			NOT = { has_global_flag = secondbalkanwar_over }
			has_war_with = BUL
		}
		available = {
			OR = {
				casualties > 50000
				surrender_progress > 0
			}
			OR = {
				is_faction_leader = yes
				SER = { has_capitulated = yes }
			}
		}

		days_remove = 3

		remove_effect = {
			BUL = { country_event = wsv_secondbalkan.13 }
			custom_effect_tooltip = secondbalkan_bulgaria_wins_tt
		}

		ai_will_do = {
			factor = -2
			# serbia losing is really bad
			modifier = {
				add = 1
				GRE = { surrender_progress > 0.5 }
			}
			modifier = {
				add = 1
				SER = { has_capitulated = yes}
			}

			# greece losing is bad
			modifier = {
				add = 1
				GRE = { surrender_progress > 0.5 }
			}
			# romania losing is bad
			modifier = {
				add = 1
				GRE = { surrender_progress > 0.5 }
			}
			# tons of casualties is bad
			modifier = {
				add = 1
				any_country = {
					is_in_faction_with = SER 
					has_war_with = BUL
					casualties > 100000
				}
			}
		}
	}
}
